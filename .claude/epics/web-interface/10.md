### Middleware Implementation
```php
// app/Http/Middleware/SetLocale.php
class SetLocale
{
    public function handle($request, Closure $next)
    {
        $locale = $this->getLocaleFromUrl($request) 
               ?? $this->getLocaleFromSession($request)
               ?? $this->getLocaleFromBrowser($request)
               ?? config('app.fallback_locale');
        
        if (!in_array($locale, config('app.available_locales'))) {
            $locale = config('app.fallback_locale');
        }
        
        App::setLocale($locale);
        Carbon::setLocale($locale);
        
        return $next($request);
    }
    
    private function getLocaleFromUrl($request): ?string
    {
        $segments = $request->segments();
        $locale = $segments[0] ?? null;
        
        return in_array($locale, config('app.available_locales')) ? $locale : null;
    }
}
```

### Blade Template Helpers
```php
// app/Helpers/LocalizationHelper.php
class LocalizationHelper
{
    public static function getCurrentLocale(): string
    {
        return App::getLocale();
    }
    
    public static function getAvailableLocales(): array
    {
        return config('localization.supported_locales');
    }
    
    public static function getLocalizedUrl(string $locale, string $route = null): string
    {
        $route = $route ?? Route::currentRouteName();
        $parameters = Route::current()->parameters();
        $parameters['locale'] = $locale;
        
        return route($route, $parameters);
    }
    
    public static function formatNumber(float $number, int $decimals = 0): string
    {
        $locale = App::getLocale();
        
        return match ($locale) {
            'ko' => number_format($number, $decimals),
            'en' => number_format($number, $decimals, '.', ','),
            default => number_format($number, $decimals)
        };
    }
    
    public static function formatDate(Carbon $date, string $format = null): string
    {
        $locale = App::getLocale();
        $format = $format ?? ($locale === 'ko' ? 'Y년 m월 d일' : 'M d, Y');
        
        return $date->format($format);
    }
}
```

### Language Detection
```php
// app/Http/Controllers/LocaleController.php
class LocaleController extends Controller
{
    public function detectAndRedirect(Request $request)
    {
        $preferredLocale = $this->detectPreferredLocale($request);
        
        return redirect()->route('home', ['locale' => $preferredLocale]);
    }
    
    private function detectPreferredLocale(Request $request): string
    {
        // 1. 세션에서 확인
        if ($request->session()->has('locale')) {
            $sessionLocale = $request->session()->get('locale');
            if (in_array($sessionLocale, config('app.available_locales'))) {
                return $sessionLocale;
            }
        }
        
        // 2. 브라우저 언어 감지
        if (config('localization.use_browser_language')) {
            $browserLocale = $this->getBrowserLocale($request);
            if ($browserLocale) {
                return $browserLocale;
            }
        }
        
        // 3. 기본 언어
        return config('app.locale');
    }
    
    private function getBrowserLocale(Request $request): ?string
    {
        $acceptLanguage = $request->server('HTTP_ACCEPT_LANGUAGE');
        
        if (!$acceptLanguage) {
            return null;
        }
        
        foreach (explode(',', $acceptLanguage) as $lang) {
            $locale = substr(trim($lang), 0, 2);
            if (in_array($locale, config('app.available_locales'))) {
                return $locale;
            }
        }
        
        return null;
    }
}
```
