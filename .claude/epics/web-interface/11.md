#### Button Component
```html
<!-- resources/views/components/button.blade.php -->
@props([
    'variant' => 'primary',
    'size' => 'md',
    'type' => 'button',
    'disabled' => false,
    'loading' => false,
    'icon' => null,
    'iconPosition' => 'left'
])

@php
$baseClasses = 'inline-flex items-center justify-center font-medium rounded-md transition duration-150 ease-in-out focus:outline-none focus:ring-2 focus:ring-offset-2 disabled:opacity-50 disabled:cursor-not-allowed';

$variants = [
    'primary' => 'bg-primary-600 hover:bg-primary-700 text-white focus:ring-primary-500',
    'secondary' => 'bg-gray-200 hover:bg-gray-300 text-gray-900 focus:ring-gray-500 dark:bg-gray-700 dark:hover:bg-gray-600 dark:text-gray-100',
    'success' => 'bg-green-600 hover:bg-green-700 text-white focus:ring-green-500',
    'danger' => 'bg-red-600 hover:bg-red-700 text-white focus:ring-red-500',
    'outline' => 'border border-gray-300 bg-white hover:bg-gray-50 text-gray-700 focus:ring-primary-500 dark:border-gray-600 dark:bg-gray-800 dark:hover:bg-gray-700 dark:text-gray-100'
];

$sizes = [
    'sm' => 'px-3 py-1.5 text-sm',
    'md' => 'px-4 py-2 text-sm',
    'lg' => 'px-6 py-3 text-base',
    'xl' => 'px-8 py-4 text-lg'
];

$classes = $baseClasses . ' ' . ($variants[$variant] ?? $variants['primary']) . ' ' . ($sizes[$size] ?? $sizes['md']);
@endphp

<button 
    type="{{ $type }}" 
    {{ $attributes->merge(['class' => $classes]) }}
    @if($disabled || $loading) disabled @endif
>
    @if($loading)
        <x-icon name="spinner" class="animate-spin h-4 w-4 {{ $slot->isNotEmpty() ? 'mr-2' : '' }}" />
    @elseif($icon && $iconPosition === 'left')
        <x-icon :name="$icon" class="h-4 w-4 {{ $slot->isNotEmpty() ? 'mr-2' : '' }}" />
    @endif
    
    {{ $slot }}
    
    @if($icon && $iconPosition === 'right')
        <x-icon :name="$icon" class="h-4 w-4 {{ $slot->isNotEmpty() ? 'ml-2' : '' }}" />
    @endif
</button>
```

#### Input Component
```html
<!-- resources/views/components/input.blade.php -->
@props([
    'type' => 'text',
    'label' => null,
    'error' => null,
    'help' => null,
    'required' => false,
    'icon' => null,
    'iconPosition' => 'left'
])

@php
$inputClasses = 'block w-full rounded-md border-gray-300 dark:border-gray-600 dark:bg-gray-700 dark:text-white shadow-sm focus:border-primary-500 focus:ring-primary-500 sm:text-sm';

if ($error) {
    $inputClasses = 'block w-full rounded-md border-red-300 dark:border-red-600 text-red-900 dark:text-red-100 placeholder-red-300 focus:border-red-500 focus:ring-red-500 sm:text-sm';
}

if ($icon) {
    $inputClasses .= $iconPosition === 'left' ? ' pl-10' : ' pr-10';
}
@endphp

<div>
    @if($label)
        <label class="block text-sm font-medium text-gray-700 dark:text-gray-300 mb-1">
            {{ $label }}
            @if($required)
                <span class="text-red-500">*</span>
            @endif
        </label>
    @endif
    
    <div class="relative">
        @if($icon && $iconPosition === 'left')
            <div class="absolute inset-y-0 left-0 pl-3 flex items-center pointer-events-none">
                <x-icon :name="$icon" class="h-5 w-5 text-gray-400" />
            </div>
        @endif
        
        <input 
            type="{{ $type }}"
            {{ $attributes->merge(['class' => $inputClasses]) }}
        />
        
        @if($icon && $iconPosition === 'right')
            <div class="absolute inset-y-0 right-0 pr-3 flex items-center pointer-events-none">
                <x-icon :name="$icon" class="h-5 w-5 text-gray-400" />
            </div>
        @endif
    </div>
    
    @if($error)
        <p class="mt-1 text-sm text-red-600 dark:text-red-400">{{ $error }}</p>
    @elseif($help)
        <p class="mt-1 text-sm text-gray-500 dark:text-gray-400">{{ $help }}</p>
    @endif
</div>
```

### Progress Bar Component
```html
<!-- resources/views/components/progress.blade.php -->
@props([
    'value' => 0,
    'max' => 100,
    'color' => 'primary',
    'size' => 'md',
    'showPercent' => false,
    'animated' => false
])

@php
$percentage = min(100, max(0, ($value / $max) * 100));

$colors = [
    'primary' => 'bg-primary-600',
    'success' => 'bg-green-600',
    'warning' => 'bg-yellow-600',
    'danger' => 'bg-red-600'
];

$sizes = [
    'sm' => 'h-1',
    'md' => 'h-2',
    'lg' => 'h-3'
];

$colorClass = $colors[$color] ?? $colors['primary'];
$sizeClass = $sizes[$size] ?? $sizes['md'];
@endphp

<div>
    @if($showPercent)
        <div class="flex justify-between text-sm text-gray-600 dark:text-gray-400 mb-1">
            <span>{{ __('ui.status.progress') }}</span>
            <span>{{ round($percentage) }}%</span>
        </div>
    @endif
    
    <div class="w-full bg-gray-200 dark:bg-gray-700 rounded-full {{ $sizeClass }}">
        <div 
            class="{{ $colorClass }} {{ $sizeClass }} rounded-full transition-all duration-300 ease-out {{ $animated ? 'animate-pulse' : '' }}"
            style="width: {{ $percentage }}%"
        ></div>
    </div>
</div>
```
